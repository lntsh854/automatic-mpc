1function [AutoMATiC_DMC_Ke, AutoMATiC_DMC_Ku, AutoMATiC_DMC_D, AutoMATiC_DMC_ny, AutoMATiC_DMC_nu]= dmc_a_init_coder(S,N,Nu,lambda,psi)
 2%% DOCUMENTATION
 3% This algorithm calculates the control signal as constant. This constant
 4% may be different for each control signal. To get control value of 2 for
 5% first contol signal and 10 for second one, "control_value" variable must
 6% be defined as follows
 7%
 8% control_value = [2;10];
 9%
10%--------------------------------------------------------------------------
11% Author: Patryk Chaber
12% Date:   19.04.2018
13%-----------------------------------------
14AutoMATiC_DMC_ny = size(S,1);
15AutoMATiC_DMC_nu = size(S,2);
16AutoMATiC_DMC_D  = size(S,3); 
17
18AutoMATiC_DMC_Lambda = eye(Nu*AutoMATiC_DMC_nu)*lambda;
19AutoMATiC_DMC_Psi = eye(N*AutoMATiC_DMC_ny)*psi;
20
21AutoMATiC_DMC_Mtmp = cell(N,Nu);
22for AutoMATiC_DMC_row = 1:N
23   for AutoMATiC_DMC_col = 1:Nu
24        if(AutoMATiC_DMC_row-AutoMATiC_DMC_col+1 >= 1)
25            AutoMATiC_DMC_Mtmp{AutoMATiC_DMC_row,AutoMATiC_DMC_col} = S(:,:,min(AutoMATiC_DMC_row-AutoMATiC_DMC_col+1,AutoMATiC_DMC_D));
26        else
27            AutoMATiC_DMC_Mtmp{AutoMATiC_DMC_row,AutoMATiC_DMC_col} = zeros(size(S(:,:,1)));
28        end
29   end
30end
31AutoMATiC_DMC_M=zeros(N*AutoMATiC_DMC_ny,Nu*AutoMATiC_DMC_nu);
32itx = 1;
33ity = 1;
34for i = 1:N
35    for j = 1:AutoMATiC_DMC_ny
36        ity = 1;
37        for k = 1:Nu
38            for l = 1:AutoMATiC_DMC_nu
39                mattmp = AutoMATiC_DMC_Mtmp{i,k};
40                AutoMATiC_DMC_M(itx,ity) = mattmp(j,l);
41                ity = ity+1;
42            end
43        end
44        itx = itx+1;
45    end
46end
47AutoMATiC_DMC_D_1 = AutoMATiC_DMC_D-1;
48% Macierz Mp
49AutoMATiC_DMC_Mptmp = cell(N,AutoMATiC_DMC_D_1);
50for AutoMATiC_DMC_row = 1:N
51   for AutoMATiC_DMC_col = 1:AutoMATiC_DMC_D_1
52        AutoMATiC_DMC_Mptmp{AutoMATiC_DMC_row,AutoMATiC_DMC_col} = S(:,:,min(AutoMATiC_DMC_row+AutoMATiC_DMC_col,AutoMATiC_DMC_D)) - S(:,:,AutoMATiC_DMC_col);
53   end
54end
55
56AutoMATiC_DMC_Mp=zeros(N*AutoMATiC_DMC_ny,(AutoMATiC_DMC_D-1)*AutoMATiC_DMC_nu);
57itx = 1;
58ity = 1;
59for i = 1:N
60    for j = 1:AutoMATiC_DMC_ny
61        ity = 1;
62        for k = 1:AutoMATiC_DMC_D_1
63            for l = 1:AutoMATiC_DMC_nu
64                mattmp = AutoMATiC_DMC_Mptmp{i,k};
65                AutoMATiC_DMC_Mp(itx,ity) = mattmp(j,l);
66                ity = ity+1;
67            end
68        end
69        itx = itx+1;
70    end
71end
72
73% Macierz K
74AutoMATiC_DMC_K = (AutoMATiC_DMC_M'*AutoMATiC_DMC_Psi*AutoMATiC_DMC_M+AutoMATiC_DMC_Lambda)^(-1)*AutoMATiC_DMC_M'*AutoMATiC_DMC_Psi;
75
76%% Macierze dla wersji minimalistycznej algorytmu
77AutoMATiC_DMC_Ku=AutoMATiC_DMC_K(1:AutoMATiC_DMC_nu,:)*AutoMATiC_DMC_Mp;
78
79AutoMATiC_DMC_Ke=zeros(AutoMATiC_DMC_nu,AutoMATiC_DMC_ny);
80for AutoMATiC_DMC_n=1:AutoMATiC_DMC_nu
81    for AutoMATiC_DMC_m=1:AutoMATiC_DMC_ny
82        AutoMATiC_DMC_Ke(AutoMATiC_DMC_n,AutoMATiC_DMC_m) = sum(AutoMATiC_DMC_K(AutoMATiC_DMC_n,AutoMATiC_DMC_m:AutoMATiC_DMC_ny:end));
83    end
84end
